maxIndex(float[] a) =>
    if array.size(a) == 0
        na
    else
        float m = array.get(a, 0)
        int idx = 0
        for i = 1 to array.size(a) - 1
            float v = array.get(a, i)
            if v > m
                m := v
                idx := i
        idx

method initialize(VolumeProfileEngine this) =>
    this.hPrices := array.new_float()
    this.lPrices := array.new_float()
    this.volumes := array.new_float()
    this.minProfilePrice := na
    this.maxProfilePrice := na
    this.priceStepSize := na
    this.lastPocPrice := na
    this.lastVahPrice := na
    this.lastValPrice := na
    this

method calculatePriceBucket(VolumeProfileEngine this, float price) =>
    float raw = math.floor((price - this.minProfilePrice) / this.priceStepSize)
    math.max(0, math.min(int(raw), FINAL_ROWS - 1))

method getBucketCenterPrice(VolumeProfileEngine this, int bucketIndex) =>
    (bucketIndex + 0.5) * this.priceStepSize + this.minProfilePrice

method storeBatchBars(VolumeProfileEngine this, float[] batchHighs, float[] batchLows, float[] batchVols) =>
    if array.size(batchHighs) > 0
        int currentSize = array.size(this.hPrices)
        int roomLeft = math.max(0, MAX_ARRAY_SIZE - currentSize)
        int barsToStore = math.min(array.size(batchHighs), roomLeft)
        if barsToStore > 0
            for i = 0 to barsToStore - 1
                float h = array.get(batchHighs, i)
                float l = array.get(batchLows, i)
                float v = array.get(batchVols, i)
                if not na(h) and not na(l) and not na(v)
                    array.push(this.hPrices, h)
                    array.push(this.lPrices, l)
                    array.push(this.volumes, v)
                    this.minProfilePrice := na(this.minProfilePrice) ? l : math.min(this.minProfilePrice, l)
                    this.maxProfilePrice := na(this.maxProfilePrice) ? h : math.max(this.maxProfilePrice, h)

method storeSingleBar(VolumeProfileEngine this, float h, float l, float v) =>
    if array.size(this.hPrices) < MAX_ARRAY_SIZE
        array.push(this.hPrices, h)
        array.push(this.lPrices, l)
        array.push(this.volumes, v)
        this.minProfilePrice := na(this.minProfilePrice) ? l : math.min(this.minProfilePrice, l)
        this.maxProfilePrice := na(this.maxProfilePrice) ? h : math.max(this.maxProfilePrice, h)

method createTwoTonePolyline(VolumeProfileEngine this, int[] barPositions, int profileStartBar, float stepSize, float rowHeight,
     color lowerColor, color upperColor) =>
    lowerPts = array.new<chart.point>()
    upperPts = array.new<chart.point>()
    float halfH = rowHeight / 2.0
    float heightGap = math.max(0.0, (stepSize - rowHeight) / 2)

    for [idx, barPos] in barPositions
        if barPos > profileStartBar
            float base = idx * stepSize + this.minProfilePrice + heightGap
            float lowerBottom = base
            float lowerTop    = base + halfH
            float upperBottom = lowerTop
            float upperTop    = base + rowHeight

            array.push(lowerPts, chart.point.from_index(profileStartBar, lowerBottom))
            array.push(lowerPts, chart.point.from_index(barPos,          lowerBottom))
            array.push(lowerPts, chart.point.from_index(barPos,          lowerTop))
            array.push(lowerPts, chart.point.from_index(profileStartBar, lowerTop))

            array.push(upperPts, chart.point.from_index(profileStartBar, upperBottom))
            array.push(upperPts, chart.point.from_index(barPos,          upperBottom))
            array.push(upperPts, chart.point.from_index(barPos,          upperTop))
            array.push(upperPts, chart.point.from_index(profileStartBar, upperTop))

    polyline pLower = na
    polyline pUpper = na
    if array.size(lowerPts) > 1
        pLower := polyline.new(lowerPts, closed=true, line_color=lowerColor, fill_color=lowerColor)
    if array.size(upperPts) > 1
        pUpper := polyline.new(upperPts, closed=true, line_color=upperColor, fill_color=upperColor)
    [pLower, pUpper]

method processAndRender(VolumeProfileEngine this, int profileStartBar, int profileEndBar, bool isActiveProfile) =>
    if array.size(this.hPrices) > 0 and profileEndBar > profileStartBar
        float priceRange = math.max(this.maxProfilePrice - this.minProfilePrice, syminfo.mintick)
        float rawStep = priceRange / FINAL_ROWS
        float minStep = syminfo.mintick * finalVpMinTickMult
        this.priceStepSize := math.max(rawStep, minStep)

        float[] volumeDistribution = array.new_float(FINAL_ROWS, 0)
        int count = array.size(this.hPrices)

        for i = 0 to count - 1
            float pHigh = array.get(this.hPrices, i)
            float pLow  = array.get(this.lPrices, i)
            float pVol  = array.get(this.volumes, i)

            int startBucket = this.calculatePriceBucket(pLow)
            int endBucket   = this.calculatePriceBucket(pHigh)
            if endBucket < startBucket
                int tmp = startBucket
                startBucket := endBucket
                endBucket := tmp

            int bucketsSpanned = endBucket - startBucket + 1
            float volumePerBucket = bucketsSpanned > 0 ? pVol / bucketsSpanned : 0.0
            for bucketIndex = startBucket to endBucket
                array.set(volumeDistribution, bucketIndex, array.get(volumeDistribution, bucketIndex) + volumePerBucket)

        float maxVolumeInProfile = array.max(volumeDistribution)
        if maxVolumeInProfile > 0
            int profileBarLength  = profileEndBar - profileStartBar
            int maxHistogramWidth = math.round(profileBarLength * MAX_WIDTH / 100)

            int[] nonVaPositions = array.new_int()
            int[] vaPositions    = array.new_int()

            int pocBucketIndex = maxIndex(volumeDistribution)
            float totalProfileVolume = array.sum(volumeDistribution)
            float valueAreaThresholdVol = totalProfileVolume * VALUE_AREA_THRESHOLD / 100
            float currentValueAreaVolume = array.get(volumeDistribution, pocBucketIndex)
            int valueAreaHighBucket = pocBucketIndex
            int valueAreaLowBucket  = pocBucketIndex
            int loopGuard = 0

            while currentValueAreaVolume < valueAreaThresholdVol and loopGuard < FINAL_ROWS
                loopGuard += 1
                bool canUp = valueAreaHighBucket < FINAL_ROWS - 1
                bool canDn = valueAreaLowBucket > 0
                  
                if not canUp and not canDn
                    break
                      
                float vUp = canUp ? array.get(volumeDistribution, valueAreaHighBucket + 1) : 0
                float vDn = canDn ? array.get(volumeDistribution, valueAreaLowBucket - 1) : 0
                  
                if (vUp >= vDn and canUp)
                    currentValueAreaVolume += vUp
                    valueAreaHighBucket += 1
                else if canDn
                    currentValueAreaVolume += vDn
                    valueAreaLowBucket -= 1
                else
                    break

            float pocPrice = this.getBucketCenterPrice(pocBucketIndex)
            float valueAreaHighPrice = (valueAreaHighBucket + 1) * this.priceStepSize + this.minProfilePrice
            float valueAreaLowPrice  = valueAreaLowBucket * this.priceStepSize + this.minProfilePrice
            this.lastPocPrice := pocPrice
            this.lastVahPrice := valueAreaHighPrice
            this.lastValPrice := valueAreaLowPrice

            bool profileSpanDrawable = (profileEndBar - profileStartBar) <= MAX_DRAW_BARS_BACK
            bool renderProfileObjects = (isActiveProfile or (RENDER_COMPLETED_PROFILE and not proLightMode)) and profileSpanDrawable

            for [bucketIndex, volumeAmount] in volumeDistribution
                int barWidth = math.round(volumeAmount / maxVolumeInProfile * maxHistogramWidth)
                int barEndPosition = profileStartBar + barWidth
                bool inVA = bucketIndex >= valueAreaLowBucket and bucketIndex <= valueAreaHighBucket
                array.push(vaPositions,    inVA ? barEndPosition : profileStartBar)
                array.push(nonVaPositions, inVA ? profileStartBar : barEndPosition)

            float rowHeight = HEIGHT_PERCENTAGE_PER_ROW / 100 * this.priceStepSize

            if renderProfileObjects
                f_clearVpObjects(this)

                if SHOW_PROFILE_HIST and not proLightMode
                    [pl1, pu1] = this.createTwoTonePolyline(nonVaPositions, profileStartBar, this.priceStepSize, rowHeight, PROFILE_COLOR_LOWER, PROFILE_COLOR_UPPER)
                    this.profileHistLower := pl1
                    this.profileHistUpper := pu1
                    [pl2, pu2] = this.createTwoTonePolyline(vaPositions, profileStartBar, this.priceStepSize, rowHeight, VALUE_AREA_COLOR_LOWER, VALUE_AREA_COLOR_UPPER)
                    this.valueAreaHistLower := pl2
                    this.valueAreaHistUpper := pu2

                int OFFSET = 1
                int lineVisualEndX = profileEndBar
                if isActiveProfile and EXTEND_POC
                    lineVisualEndX := bar_index + OFFSET

                int labelX = profileStartBar
                color labelBgColor = color.new(VP_LABEL_BG_COLOR, VP_LABEL_BG_TRANS)
                bool renderVpLeftLabels = not isActiveProfile or SHOW_ACTIVE_VP_LABELS

                if SHOW_POC
                    this.pocLevelLine := line.new(profileStartBar, pocPrice, lineVisualEndX, pocPrice, color=POC_COLOR, width=POC_WIDTH, style=poc_style_str)
                    string txt = "POC"
                    if SHOW_PRICE_LABELS
                        txt += ": " + str.tostring(pocPrice, format.mintick)
                    if renderVpLeftLabels
                        this.pocLevelLabel := label.new(labelX, pocPrice, txt, xloc=xloc.bar_index, yloc=yloc.price,
                              color=labelBgColor, textcolor=POC_COLOR, style=label.style_label_up, textalign=text.align_left, size=VP_LABEL_SIZE)

                if SHOW_VAH_VAL
                    color vahLineCol = SHOW_VAH_VAL_LINES ? VAH_COLOR : color.new(VAH_COLOR, 100)
                    color valLineCol = SHOW_VAH_VAL_LINES ? VAL_COLOR : color.new(VAL_COLOR, 100)

                    this.vahLevelLine := line.new(profileStartBar, valueAreaHighPrice, lineVisualEndX, valueAreaHighPrice, color=vahLineCol, width=VAHVAL_WIDTH, style=vahval_style_str)
                    this.valLevelLine := line.new(profileStartBar, valueAreaLowPrice,  lineVisualEndX, valueAreaLowPrice,  color=valLineCol, width=VAHVAL_WIDTH, style=vahval_style_str)

                    string txtVah = "VAH"
                    string txtVal = "VAL"
                    if SHOW_PRICE_LABELS
                        txtVah += ": " + str.tostring(valueAreaHighPrice, format.mintick)
                        txtVal += ": " + str.tostring(valueAreaLowPrice,  format.mintick)

                    if renderVpLeftLabels
                        this.vahLevelLabel := label.new(labelX, valueAreaHighPrice, txtVah, xloc=xloc.bar_index, yloc=yloc.price,
                              color=labelBgColor, textcolor=VAH_COLOR, style=label.style_label_down, textalign=text.align_left, size=VP_LABEL_SIZE)
                        this.valLevelLabel := label.new(labelX, valueAreaLowPrice,  txtVal, xloc=xloc.bar_index, yloc=yloc.price,
                              color=labelBgColor, textcolor=VAL_COLOR, style=label.style_label_up,    textalign=text.align_left, size=VP_LABEL_SIZE)

                    if SHOW_BACKGROUND and SHOW_VAH_VAL_LINES
                        this.vaFill := linefill.new(this.vahLevelLine, this.valLevelLine, BACKGROUND_COLOR)

    sec < 2700  ? "D" :
     sec < 10800 ? "W" :
     sec < 86400 ? "M" : "12M"

getAutoPrecisionTF() =>
    int sec = timeframe.in_seconds(timeframe.period)
    sec <= 180   ? "1"  :
     sec <= 300  ? "1"  :
     sec <= 900  ? "5"  :
     sec <= 1800 ? "5"  :
     sec <= 3600 ? "15" :
     sec <= 14400? "60" :
     sec <= 86400? "240": "D"

    float sumWSrc = 0.0
    float sumWSrcSq = 0.0 // Added for Variance/SD Calculation

method update(VwapAccumulator this, bool reset, float w, float src) =>
    if reset
        this.sumW := 0.0
        this.sumWSrc := 0.0
        this.sumWSrcSq := 0.0
    if na(this.sumW)
        this.sumW := 0.0
    if na(this.sumWSrc)
        this.sumWSrc := 0.0
    if na(this.sumWSrcSq)
        this.sumWSrcSq := 0.0
    float ww = na(w) ? 0.0 : w
    if ww > 0 and not na(src)
        this.sumW += ww
        this.sumWSrc += ww * src
        this.sumWSrcSq += ww * src * src

method getVwap(VwapAccumulator this) =>
    nz(this.sumW, 0.0) > 0 ? nz(this.sumWSrc, 0.0) / nz(this.sumW, 0.0) : na

// New: Method to get Standard Deviation
method getStDev(VwapAccumulator this, float vwapVal) =>
    if nz(this.sumW, 0.0) > 0 and not na(vwapVal)
        float variance = (nz(this.sumWSrcSq, 0.0) / nz(this.sumW, 0.0)) - (vwapVal * vwapVal)
        math.sqrt(math.max(0, variance))
    else
        na

// f_fmt_cvd(v) — compact delta display: 亚+15K / 伦-3.2K / 纽~0
f_fmt_cvd(float v) =>
    float absV = math.abs(v)
    if absV < 1
        "~0"
    else if absV >= 1e6
        (v > 0 ? "+" : "-") + str.tostring(math.round(absV / 1e5) * 0.1, "0.0") + "M"
    else if absV >= 1e3
        (v > 0 ? "+" : "-") + str.tostring(math.round(absV / 1e2) * 0.1, "0.0") + "K"
    else
        (v > 0 ? "+" : "-") + str.tostring(math.round(absV))

updateLabel(label lb, line ln, string name, float price, color col, bool isHigh, bool isSwept) =>
    int startX = line.get_x1(ln)
    int endX   = line.get_x2(ln)
    int midX   = int(math.avg(startX, endX))
    bool isH1 = timeframe.in_seconds() == 3600
    bool isWeekPool = str.contains(name, "上周")
    bool isDayPool = str.contains(name, " 高") or str.contains(name, " 低")
    string activeSize = isWeekPool ? WEEK_LIQUIDITY_LABEL_SIZE : isDayPool ? DAY_LIQUIDITY_LABEL_SIZE : isH1 ? size.tiny : ICT_UNSWEPT_LABEL_SIZE
    string sweptSize = ICT_SWEPT_LABEL_SIZE
    string sz = isSwept ? sweptSize : activeSize
    string dispName = name
    if isH1
        if str.contains(name, "亚洲盘")
            dispName := str.replace(name, "亚洲盘", "亚")
        else if str.contains(name, "伦敦盘")
            dispName := str.replace(name, "伦敦盘", "伦")
        else if str.contains(name, "纽约盘")
            dispName := str.replace(name, "纽约盘", "纽")
    color txtCol = isSwept ? color.new(col, 40) : col
    label.set_xy(lb, midX, price)
    label.set_text(lb, dispName + ": " + str.tostring(price, format.mintick))
    label.set_size(lb, sz)
    label.set_color(lb, color.new(ICT_LABEL_BG_COLOR, ICT_LABEL_BG_TRANS))
    label.set_textcolor(lb, txtCol)
    label.set_style(lb, isHigh ? label.style_label_down : label.style_label_up)

f_cutoff_ms(int lookbackDays) =>
    int base = time - (lookbackDays * 86400 * 1000)
    if not isCrypto
        int dow = dayofweek(time, DISP_TZ)
        if dow == dayofweek.monday
            base -= 2 * 86400 * 1000
    base

f_day_name_from_time(int ts) =>
    int dow = dayofweek(ts, DISP_TZ)
    switch dow
        1 => "周日"
        2 => "周一"
        3 => "周二"
        4 => "周三"
        5 => "周四"
        6 => "周五"
        7 => "周六"
        => ""

f_remove_level_from_array(ICTLevel target) =>
    if array.size(levels) > 0 and not na(target)
        int i = array.size(levels) - 1
        while i >= 0
            ICTLevel lvl = array.get(levels, i)
            if lvl.ln == target.ln and lvl.name == target.name
                array.remove(levels, i)
                break
            i -= 1

    if not na(hiLabel)

manageSession(bool isSession, bool showSession, bool allowDraw, color sessionColor, string sessionName, int prio, SessionState st) =>
    int cutoffTime = f_cutoff_ms(ICT_LOOKBACK_DAYS)
    string shortName = sessionName == "亚洲盘" ? "亚" : sessionName == "伦敦盘" ? "伦" : "纽"
    if showSession
        bool justStarted = isSession and not isSession[1]
        if justStarted
            st.ended      := false
            st.sOpen      := open
            st.sHigh      := high
            st.sLow       := low
            st.sClose     := close
            st.sHighBar   := bar_index
            st.sLowBar    := bar_index
            st.startTime := time
            int dow = dayofweek(time, DISP_TZ)
            string dayName = switch dow
                1 => "周日"
                2 => "周一"
                3 => "周二"
                4 => "周三"
                5 => "周四"
                6 => "周五"
                7 => "周六"
                => ""
            st.dayName := dayName
            if allowDraw and st.startTime > cutoffTime
                string displayName = dayName + " " + shortName
                color bgCol = color.new(ICT_LABEL_BG_COLOR, ICT_LABEL_BG_TRANS)
                color initLineColor = SHOW_OPEN_ONLY_ACTIVE ? na : color.new(sessionColor, 0)
                // 进行中：虚线 + "(进行中)"后缀；会话收盘后才转实线、去后缀，避免把未完成会话的实时极值误读为已成立流动性
                st.lHigh := line.new(st.sHighBar, st.sHigh, bar_index, st.sHigh, color=initLineColor, width=ICT_LINE_WIDTH, style=line.style_dashed)
                st.lbHigh := label.new(bar_index, st.sHigh, displayName + " 高(进行中)", color=bgCol, textcolor=sessionColor, style=label.style_label_down, size=ICT_UNSWEPT_LABEL_SIZE)
                if SHOW_OPEN_ONLY_ACTIVE
                    label.set_text(st.lbHigh, "")
                    label.set_style(st.lbHigh, label.style_none)

// FVG 公允价值缺口（ICT 三K失衡）—— 自动标注 + 回踩入场汇合

type FVG
    box   bx
    line  ce
    float top
    float bot

    bool  htfConf

// ===== 本级 FVG 维护 + 高周期重叠确认 =====
// 本级FVG与同向HTF FVG区重叠=有高周确认(htfConf)：边框加粗加实、核对行标 FVG✓HTF
bool inBullFvg = false
bool inBearFvg = false
bool inBullFvgHtf = false
bool inBearFvgHtf = false
float bullFvgCe = na
float bearFvgCe = na
float fvgNearDist = FVG_NEAR_ATR * (na(currATR) ? 0.0 : currATR)
if SHOW_FVG and array.size(fvgList) > 0
    int fi = array.size(fvgList) - 1
    while fi >= 0
        FVG f = array.get(fvgList, fi)
        bool filled = f.isBull ? low <= f.bot : high >= f.top
        if filled
            if not na(f.bx)
                box.delete(f.bx)
            if not na(f.ce)
                line.delete(f.ce)
            array.remove(fvgList, fi)
        else
            // 高周期重叠确认：同向 + 价格区间相交
            bool htfConf = false
            if fvgHtfValid and array.size(htfFvgList) > 0
                int gj = 0
                while gj < array.size(htfFvgList)
                    FVG hz = array.get(htfFvgList, gj)
                    if hz.isBull == f.isBull and f.top >= hz.bot and f.bot <= hz.top
                        htfConf := true
                    gj += 1
            f.htfConf := htfConf
            // 入场汇合
            if f.isBull and close <= f.top and close >= f.bot
                inBullFvg := true
                inBullFvgHtf := inBullFvgHtf or htfConf
                bullFvgCe := math.avg(f.top, f.bot)
            if not f.isBull and close >= f.bot and close <= f.top
                inBearFvg := true
                inBearFvgHtf := inBearFvgHtf or htfConf
                bearFvgCe := math.avg(f.top, f.bot)
            // 画框/中线：离价近才显示；有HTF确认的边框更实更粗
            if not na(f.bx)
                float distToGap = f.isBull ? math.max(0.0, close - f.top) : math.max(0.0, f.bot - close)
                bool nearPrice = fvgNearDist <= 0.0 or distToGap <= fvgNearDist
                color baseCol = f.isBull ? FVG_BULL_COL : FVG_BEAR_COL
                box.set_right(f.bx, bar_index + FVG_EXTEND)
                box.set_bgcolor(f.bx, nearPrice ? color.new(baseCol, FVG_FILL_TRANSP) : na)
                box.set_border_color(f.bx, nearPrice ? color.new(baseCol, htfConf ? 0 : 40) : na)
                box.set_border_width(f.bx, htfConf ? 2 : 1)
                if not na(f.ce)
                    line.set_x2(f.ce, bar_index + FVG_EXTEND)
                    line.set_color(f.ce, nearPrice ? color.new(baseCol, 20) : na)
        fi -= 1
    while array.size(fvgList) > FVG_MAX_COUNT
        FVG oldf = array.shift(fvgList)
        if not na(oldf.bx)
            box.delete(oldf.bx)
        if not na(oldf.ce)
            line.delete(oldf.ce)

    float threshold = math.max(syminfo.mintick * 4, math.abs(close) * 0.00005)
    bool dup1 = not na(level) and not na(ref1) and math.abs(level - ref1) <= threshold
    bool dup2 = not na(level) and not na(ref2) and math.abs(level - ref2) <= threshold
    bool dup3 = not na(level) and not na(ref3) and math.abs(level - ref3) <= threshold
    bool dup4 = not na(level) and not na(ref4) and math.abs(level - ref4) <= threshold
    not na(level) and not (dup1 or dup2 or dup3 or dup4)
